
\frame{\frametitle{课程报告}
	\begin{enumerate}
		\item 上课时各小组的幻灯或小报告
		\item 每位同学整理一份报告
			\begin{itemize}
				\item 对已有的一个方法的深入阐述，有余力的同学可编程实验并评价
				\item 探索新的方法
				\item 针对科研中的问题，提出计算智能方法并试验
			\end{itemize}
		\item 协作网址或报告发邮件给 yxxinyuan@zju.edu.cn
	\end{enumerate}
}
